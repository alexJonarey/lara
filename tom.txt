package Controller;

import modelos.Estudiante;
import java.io.IOException;
import java.io.UnsupportedEncodingException;
import java.net.URI;
import static java.net.URLEncoder.encode;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;

/**
 *
 * @author Lenovo
 */
public class ConsumidorApi2 {
    
    private static final String API_URL = "http://localhost/soa/Controllers/api.php";
    private static final HttpClient cliente = HttpClient.newHttpClient();

    public static String obtenerEstudiantes(){
        try{
        HttpRequest peticion = HttpRequest.newBuilder()
                .uri(URI.create(API_URL)).GET().build();
        return cliente.send(peticion, HttpResponse.BodyHandlers.ofString()).body();
        
        }catch (Exception e){
            System.err.println("Error Get: " + e.getMessage());
            return "";
        }
        
    } 
    
    
    public static String insertar (Estudiante e) {
 
        try {
           String body = "txtcedula=" + encode(e.getCedula(), "UTF-8")
        + "&txtnombre=" + encode(e.getNombre(), "UTF-8")
        + "&txtapellido=" + encode(e.getApellido(), "UTF-8")
        + "&txtdirecion=" + encode(e.getDireccion(), "UTF-8")
        + "&txttelefono=" + encode(e.getTelefono(), "UTF-8");
        
        HttpRequest peticion = HttpRequest.newBuilder()
                .uri(URI.create(API_URL))
                .header("Content-Type", "application/x-www-form-urlencoded")
                .POST(HttpRequest.BodyPublishers.ofString(body)).build();
            return cliente.send(peticion, HttpResponse.BodyHandlers.ofString()).body();
        } catch (Exception ex) {
            System.err.println("Error Insertar: " + ex.getMessage());
            return "";
        }
    }
    
    
    public static String eliminar(String cedula) throws UnsupportedEncodingException{
        
         HttpRequest peticion= HttpRequest.newBuilder()
                .uri(URI.create(API_URL+"?txtcedula="+encode(cedula))).DELETE().build();
        try {
            return cliente.send(peticion, HttpResponse.BodyHandlers.ofString()).body();
        } catch (Exception ex) {
            System.err.println("Error Delete: " + ex.getMessage());
            return "";
        }
    }
    
    
    public static String actualizar(Estudiante e){
        try{
            String params = "?txtcedula=" + encode(e.getCedula())
                    + "&txtnombre=" + encode(e.getNombre())
                    + "&txtapellido=" + encode(e.getApellido())
                    + "&txtdireccion=" + encode(e.getDireccion())
                    + "&txttelefono=" + encode(e.getTelefono());
        
        HttpRequest peticion = HttpRequest.newBuilder()
                .uri(URI.create(API_URL + params))
                .PUT(HttpRequest.BodyPublishers.noBody()).build();
        
        return cliente.send(peticion, HttpResponse.BodyHandlers.ofString()).body();
        
        }catch (Exception ex) {
            System.err.println("Error Actualizar: " + ex.getMessage());
            return "";
        }
    }
    
}

package servers;

import modelos.Estudiante;
import Controller.ConsumidorApi2;
import com.google.gson.JsonArray;
import com.google.gson.JsonParser;
import java.io.IOException;
import java.io.PrintWriter;
import jakarta.servlet.ServletException;
import jakarta.servlet.annotation.WebServlet;
import jakarta.servlet.http.HttpServlet;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;

@WebServlet(name = "EstudiantesServlet2", urlPatterns = {"/estudiantes2"})
public class EstudianteServlet2 extends HttpServlet {

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        // Validar parámetros
        String cedula = request.getParameter("cedula");
        String nombre = request.getParameter("nombre");
        String apellido = request.getParameter("apellido");
        String telefono = request.getParameter("telefono");
        String direccion = request.getParameter("direccion");
        
        if (cedula == null || cedula.trim().isEmpty()) {
            response.sendError(HttpServletResponse.SC_BAD_REQUEST, "La cédula es requerida");
            return;
        }
        
        Estudiante e = new Estudiante(cedula, nombre, apellido,  direccion, telefono);
        ConsumidorApi2.insertar(e);
        response.sendRedirect(request.getContextPath() + "/estudiantes2");
    }
    
    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        String accion = request.getParameter("accion");
        
        if (accion == null || accion.isEmpty()) {
            String json = ConsumidorApi2.obtenerEstudiantes();
            JsonArray lista = JsonParser.parseString(json).getAsJsonArray();
            request.setAttribute("estudiantes", lista);
            request.getRequestDispatcher("/estudiantes2.jsp").forward(request, response);
        } else if (accion.equals("irAgregar")) {
            request.getRequestDispatcher("/agregar2.jsp").forward(request, response);
        } else if (accion.equals("editar")) {
            String cedula = request.getParameter("cedula");
            if (cedula != null && !cedula.isEmpty()) {
                request.setAttribute("cedula", cedula);
                request.getRequestDispatcher("/editar2.jsp").forward(request, response);
            } else {
                response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Cédula no proporcionada");
            }
        }
    }
    
    @Override
    protected void doPut(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        // Para usar PUT, necesitas configurar un filtro o usar _method=PUT
         String cedula = request.getParameter("cedula");
        String nombre = request.getParameter("nombre");
        String apellido = request.getParameter("apellido");
        String telefono = request.getParameter("telefono");
        String direccion = request.getParameter("direccion");
         Estudiante e = new Estudiante(cedula, nombre, apellido,  direccion, telefono);

        String respuesta = ConsumidorApi2.actualizar(e);
        response.setContentType("application/json");
        response.getWriter().write(respuesta);
    }
    
    @Override
    protected void doDelete(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        String cedula = request.getParameter("cedula");
        if (cedula == null || cedula.isEmpty()) {
            response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Cédula requerida");
            return;
        }
        String respuesta = ConsumidorApi2.eliminar(cedula);
        response.setContentType("application/json");
        response.getWriter().write(respuesta);
    }
}
