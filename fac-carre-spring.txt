package com.uta.Grupo05.controller;

import com.uta.Grupo05.entities.Facultad;
import com.uta.Grupo05.repository.carreraRepository;
import com.uta.Grupo05.repository.FacultadRepository;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import com.uta.Grupo05.entities.carrera;

@Controller
@RequestMapping("/carreras")
public class CarreraController {

    private final carreraRepository carreraRepository;
    private final FacultadRepository facultadRepository;

    public CarreraController(
            carreraRepository carreraRepository,
            FacultadRepository facultadRepository
    ) {
        this.carreraRepository = carreraRepository;
        this.facultadRepository = facultadRepository;
    }

    @GetMapping
    public String listar(Model model){

        model.addAttribute(
                "carreras",
                carreraRepository.findAll()
        );

        model.addAttribute(
                "facultades",
                facultadRepository.findAll()
        );

        return "carreras";
    }

    @PostMapping
    public String guardar(
            @RequestParam String nombre,

            @RequestParam Long facultadId

    ){

        Facultad facultad =
                facultadRepository.findById(facultadId)
                        .orElseThrow();

        carrera carrera = new carrera();

        carrera.setNombre(nombre);


        carrera.setFacultad(facultad);

        carreraRepository.save(carrera);


        return "redirect:/carreras";
    }
}
 package com.uta.Grupo05.controller;
import com.uta.Grupo05.entities.Facultad;
import com.uta.Grupo05.repository.FacultadRepository;

import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;
@Controller
@RequestMapping("/facultad")
public class FacultadController {

    private final FacultadRepository facultadRepository;

    public FacultadController(FacultadRepository facultadRepository) {
        this.facultadRepository = facultadRepository;
    }

    @GetMapping
    public String listar(Model model){

        model.addAttribute(
                "facultad",
                facultadRepository.findAll()
        );

        return "facultad";
    }

    @PostMapping
    public String guardar(
            @RequestParam String nombre,
            @RequestParam String direccion
    ){

        Facultad facultad = new Facultad();

        facultad.setNombre(nombre);
        facultad.setDireccion(direccion);

        facultadRepository.save(facultad);

        return "redirect:/facultad";
    }
}
 
package com.uta.Grupo05.entities;

import jakarta.persistence.*;
import lombok.*;

@Entity
@Table(name = "carrera")
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class carrera {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    private String nombre;


    @ManyToOne
    @JoinColumn(name = "facultad_id")
    private Facultad facultad;
} 
package com.uta.Grupo05.entities;

import jakarta.persistence.*;
import lombok.*;

import java.util.List;

@Entity
@Table(name = "facultad")
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class Facultad {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Integer id;

    private String nombre;

    private String direccion;

    @OneToMany(mappedBy = "facultad")
    private List<carrera> carreras;
}
